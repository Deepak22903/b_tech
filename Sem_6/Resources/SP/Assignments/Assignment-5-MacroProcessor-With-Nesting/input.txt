MACRO ADD NUM1, NUM2
    LOAD NUM1
    ADD NUM2
    STORE RESULT
MEND
MACRO MULTIPLY NUM1, NUM2
    ADD #0
    STORE RESULT
    ADD #0
    STORE COUNT
    WHILE COUNT, NUM1
        ADD NUM2
        ADD #1
        STORE COUNT
    MEND
START
    ADD 10, 20
    MULTIPLY 5, 3
    END
